## Install

```bash
npm install --save survezy_widget
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'survezy_widget'
import 'survezy_widget/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
